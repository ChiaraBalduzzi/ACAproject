# Total times 4 cores
# Version 1 break 
# Version 1 task 
# Seriale 1

#Misurazioni: per S 10^5 P 10^3 - S 10^6 P 10^3 - S 10^7 P 10^3 - S 10^7 P 10^4


serial = np.array([
    [100000000, 0.00581 ],
    [1000000000,0.05085 ],
    [10000000000,0.53645 ],
    [100000000000,12.90995 ]

])

v1break = np.array([
    [100000000,0.00371],
    [1000000000,0.03421],
    [10000000000,0.32978],
    [100000000000,5.75996]
])

v1task= np.array([
    [100000000,0.01111],
    [1000000000,0.07921],
    [10000000000,0.75030],
    [100000000000,17.91615] # VALORE STRANO!!
])


x, y = serial.T
plt.figure(figsize=(25,8))
plt.plot(x,y, marker="o",  label="serial")

x, y = v1break.T
plt.plot(x,y, marker="*",  label="V1")

x, y = v1task.T
plt.plot(x,y, marker="+",  label="V1 task")

plt.grid()
plt.xlabel("Slen x Plen")
plt.ylabel("Total Time [s]")
plt.title("Serial vs Parallel 8 cores V1")
plt.legend(loc="upper left")
plt.show()
